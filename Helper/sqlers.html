SQL для вывода самых новых комментов по rent_id(только по одному)
SELECT DISTINCT ON (rent_id) rent_id, text, date_created FROM carrent.rent_comment ORDER BY rent_id, date_created DESC

ceil(EXTRACT(epoch from date_finish::timestamp - date_start::timestamp)/(3600*24)) AS "DaysCount"
